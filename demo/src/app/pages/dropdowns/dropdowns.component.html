Component API

Complete API reference for inputs, outputs, and interfaces.

Input Properties

Input Properties

@Input() options: DropdownOption[]        // Array of dropdown options
@Input() placeholder: string               // Placeholder text
@Input() searchable: boolean = false      // Enable search functionality
@Input() multiSelect: boolean = false     // Enable multi-select mode
@Input() disabled: boolean = false        // Disable the dropdown
@Input() size: 'sm' | 'md' | 'lg' = 'md'  // Dropdown size
@Input() maxHeight: string = '300px'      // Max height of dropdown panel
@Input() error: boolean = false           // Show error state
@Input() clearable: boolean = true        // Show clear button
@Input() showCheckboxes: boolean = false  // Show checkboxes for single select

@Output() selectionChange                 // Emits when selection changes
@Output() searchChange                    // Emits when search term changes

DropdownOption Interface

interface DropdownOption {{'{' }}
  value: any;              // Option value
  label: string;           // Display label
  disabled?: boolean;      // Disable option
  icon?: string;           // Icon/emoji
  description?: string;    // Additional description
{{'}' }}
