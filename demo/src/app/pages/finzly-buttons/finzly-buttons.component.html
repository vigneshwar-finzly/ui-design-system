Component API

Complete API reference for all button components with inputs, outputs, and usage examples.

Finzly Back Button API

Input Properties

@Input() text: string = 'Back'                      // Button text
@Input() showIcon: boolean = true                   // Show icon
@Input() icon: string = 'refresh'                   // Icon name
@Input() disabled: boolean = false                   // Disable button
@Input() size: 'sm' | 'md' | 'lg' = 'md'          // Button size
@Input() type: 'button' | 'submit' | 'reset' = 'button'
@Input() variant: 'black' | 'default' = 'black'    // Button variant

@Output() clicked                                    // Click event emitter

Finzly Custom Button API

Input Properties

@Input() text: string = 'Button'                     // Button text
@Input() icon: string = ''                           // Icon name
@Input() showIcon: boolean = true                   // Show icon
@Input() disabled: boolean = false                   // Disable button
@Input() loading: boolean = false                    // Loading state
@Input() size: 'sm' | 'md' | 'lg' = 'md'          // Button size
@Input() variant: 'primary'|'outline'|'secondary'|'danger'|'success'|'warning' = 'primary'
@Input() type: 'button' | 'submit' | 'reset' = 'button'
@Input() iconPosition: 'left' | 'right' = 'left'   // Icon position

@Output() clicked                                    // Click event emitter

Finzly Dropdown Button API

Input Properties

@Input() text: string = 'Action'                     // Button text
@Input() icon: string = ''                           // Icon name
@Input() options: DropdownButtonOption[] = []        // Dropdown options
@Input() selectedOption: DropdownButtonOption | null = null
@Input() disabled: boolean = false                   // Disable button
@Input() size: 'sm' | 'md' | 'lg' = 'md'          // Button size
@Input() variant: 'primary'|'outline'|'secondary'|'success'|'warning'|'danger' = 'primary'
@Input() type: 'button' | 'submit' | 'reset' = 'button'
@Input() placeholder: string = 'Select...'          // Placeholder text

@Output() buttonClicked                              // Main button click
@Output() optionSelected                              // Option selection


DropdownButtonOption Interface

interface DropdownButtonOption {{'{' }}
  label: string;           // Option label
  value: any;              // Option value
  icon?: string;           // Optional icon
  disabled?: boolean;      // Disable option
{{'}' }}
