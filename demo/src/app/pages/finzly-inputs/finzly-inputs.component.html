Finzly Input API

Input Properties

@Input() label: string                              // Label text
@Input() placeholder: string = ''                    // Placeholder text
@Input() required: boolean = false                   // Required field
@Input() disabled: boolean = false                   // Disable input
@Input() type: 'text'|'email'|'password'|'number'|'tel'|'url' = 'text'
@Input() size: 'sm' | 'md' | 'lg' = 'md'            // Input size
@Input() state: 'default'|'success'|'error'|'warning' = 'default'
@Input() helperText: string = ''                     // Helper text
@Input() errorText: string = ''                      // Error message
@Input() maxLength?: number                          // Max character length
@Input() minLength?: number                          // Min character length
@Input() pattern?: string                            // Validation pattern
@Input() autocomplete?: string                       // Autocomplete attribute

// Supports ngModel and formControlName (ControlValueAccessor)

Finzly Amount Input API

Input Properties

@Input() label: string = ''                          // Label text
@Input() placeholder: string = '0.00'                // Placeholder
@Input() required: boolean = false                   // Required field
@Input() disabled: boolean = false                   // Disable input
@Input() size: 'sm' | 'md' | 'lg' = 'md'            // Input size
@Input() state: 'default'|'success'|'error'|'warning' = 'default'
@Input() helperText: string = ''                     // Helper text
@Input() errorText: string = ''                      // Error message
@Input() prefix: string = '$'                        // Currency prefix
@Input() suffix: string = ''                          // Currency suffix
@Input() decimalPrecision: number = 2                // Decimal places
@Input() allowNegative: boolean = true               // Allow negative values
@Input() maxValue?: number                           // Maximum value
@Input() minValue?: number                           // Minimum value
@Input() showPrefix: boolean = true                  // Show prefix
@Input() showSuffix: boolean = false                 // Show suffix

// Supports ngModel and formControlName (ControlValueAccessor)

Finzly Textarea API

Input Properties

@Input() label: string = ''                          // Label text
@Input() placeholder: string = ''                    // Placeholder
@Input() required: boolean = false                   // Required field
@Input() disabled: boolean = false                   // Disable textarea
@Input() rows: number = 4                            // Initial rows
@Input() maxLength?: number                          // Max character length
@Input() minLength?: number                          // Min character length
@Input() state: 'default'|'success'|'error'|'warning' = 'default'
@Input() helperText: string = ''                     // Helper text
@Input() errorText: string = ''                      // Error message
@Input() autoResize: boolean = true                  // Auto-resize height
@Input() showCharCount: boolean = false              // Show character counter

// Supports ngModel and formControlName (ControlValueAccessor)

Finzly Checkbox API

Input Properties

@Input() label: string = ''                          // Label text
@Input() disabled: boolean = false                   // Disable checkbox
@Input() id: string                                  // Unique ID (auto-generated)
@Input() helperText: string = ''                     // Helper text
@Input() indeterminate: boolean = false               // Indeterminate state

// Supports ngModel and formControlName (ControlValueAccessor)
// Value: boolean

Finzly Radio Button API

Input Properties

@Input() label: string = ''                          // Label text
@Input() value: any                                  // Radio value
@Input() name: string                                // Group name (auto-generated)
@Input() disabled: boolean = false                   // Disable radio
@Input() id: string                                  // Unique ID (auto-generated)
@Input() helperText: string = ''                     // Helper text

// Supports ngModel and formControlName (ControlValueAccessor)
// Use same 'name' for radio group

Finzly Toggle API

Input Properties

@Input() label: string = ''                          // Label text
@Input() disabled: boolean = false                   // Disable toggle
@Input() id: string                                  // Unique ID (auto-generated)
@Input() helperText: string = ''                     // Helper text
@Input() onLabel: string = ''                        // Label when ON
@Input() offLabel: string = ''                        // Label when OFF

// Supports ngModel and formControlName (ControlValueAccessor)
// Value: boolean
