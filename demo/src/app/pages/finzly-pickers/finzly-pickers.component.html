Finzly Date Picker API

Input Properties

@Input() label: string = ''                          // Label text
@Input() placeholder: string = 'Select date'        // Placeholder
@Input() required: boolean = false                   // Required field
@Input() disabled: boolean = false                   // Disable picker
@Input() minDate?: Date                              // Minimum selectable date
@Input() maxDate?: Date                              // Maximum selectable date
@Input() state: 'default'|'success'|'error'|'warning' = 'default'
@Input() helperText: string = ''                     // Helper text
@Input() errorText: string = ''                       // Error message
@Input() format: string = 'MM/dd/yyyy'               // Date format
@Input() timezone?: string                           // Timezone

// Supports ngModel and formControlName (ControlValueAccessor)
// Value: Date | null

// Value: string | null (format: 'HH:mm' or 'hh:mm AM/PM')

Finzly Custom Date Picker API

Input Properties

@Input() label: string = ''                          // Label text
@Input() placeholder: string = 'Select date range' // Placeholder
@Input() required: boolean = false                   // Required field
@Input() disabled: boolean = false                   // Disable picker
@Input() minDate?: Date                              // Minimum selectable date
@Input() maxDate?: Date                              // Maximum selectable date
@Input() state: 'default'|'success'|'error'|'warning' = 'default'
@Input() helperText: string = ''                     // Helper text
@Input() errorText: string = ''                       // Error message
@Input() presets?: CustomDateConfig[]                // Date presets
@Input() showPresets: boolean = true                 // Show preset buttons
@Input() allowCustom: boolean = true                  // Allow custom date selection

// Supports ngModel and formControlName (ControlValueAccessor)
// Value: Date | null


CustomDateConfig Interface

interface CustomDateConfig {{'{' }}
  preset: DatePreset;               // Preset type
  operator: '+' | '-';              // Addition or subtraction
  value: number;                    // Value amount
  unit: 'day'|'week'|'month'|'year' // Time unit
{{'}' }}

type DatePreset = 'TODAY' | 'FDW' | 'LDW' | 'FDM' | 'LDM' | 'FDY' | 'LDY' | 'CUSTOM'
